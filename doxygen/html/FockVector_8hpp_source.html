191 FockVector<F> conjugateTranspose() const {
192 FockVector<F> result;
193 for (size_t i(0); i < componentTensors.size(); ++i) {
194 size_t order(getIndices(i).length() / 2);
195 std::vector<int> transposedLens(get(i)->lens, get(i)->lens + 2*order);
196 std::rotate(
197 transposedLens.begin(),
198 transposedLens.begin() + order,
199 transposedLens.begin() + 2*order
200 );
201 result.componentTensors.push_back(
202 NEW(Tensor<F>,
203 transposedLens.size(), transposedLens.data(),
204 get(i)->sym, *get(i)->wrld,
205 (std::string(get(i)->get_name()) + "*").c_str()
206 )
207 );
208 result.componentIndices.push_back(
209 getIndices(i).substr(order, 2*order) + getIndices(i).substr(0, order)
210 );
211 CTF::Univar_Function<F> fConj(sisi4s::conj<F>);
212 result.get(i)->sum(
213 1.0, *get(i), getIndices(i).c_str(),
214 0.0, result.getIndices(i).c_str(), fConj

288 void fromIndex(
289 const size_t index, size_t &component, size_t &componentIndex
290 ) const {
291 component = 0;
292 size_t base(0);
293 while (component < indexEnds.size()) {
294 if (index < indexEnds[component]) break;
295 base = indexEnds[component++];

297 if (component >= indexEnds.size()) {
298 throw new EXCEPTION("Index out bounds");
299 }
300 componentIndex = index - base;

307 std::vector<std::pair<size_t,F>> readLocal() const {
308 size_t elementsCount(0);
309 std::vector<std::pair<size_t,F>> elements;
310 for (size_t i(0); i < componentTensors.size(); ++i) {
311 size_t componentValuesCount;
312 size_t *componentIndices;
313 F *componentValues;
314 get(i)->read_local(
315 reinterpret_cast<int64_t *>(&componentValuesCount),
316 reinterpret_cast<int64_t **>(&componentIndices),
317 &componentValues
318 );
319
320 elements.resize(elementsCount+componentValuesCount);
321 for (size_t k(0); k < componentValuesCount; ++k) {
322 // translate index within component tensor to FockVector index
323 elements[elementsCount+k].first = getIndex(i, componentIndices[k]);
324 elements[elementsCount+k].second = componentValues[k];
325 }
326 elementsCount += componentValuesCount;
327 free(componentIndices);
328 free(componentValues);
329 }
330 return elements;
331 }

337 void write(const std::vector<std::pair<size_t,F>> &elements) {
338 // vectors to contain indices and values for each component tensor
339 std::vector<std::vector<size_t>> tensorIndices(componentTensors.size());
340 std::vector<std::vector<F>> tensorValues(componentTensors.size());
341
342 for (size_t k(0); k < elements.size(); ++k) {
343 size_t component;
344 size_t componentIndex;
345 fromIndex(elements[k].first, component, componentIndex);
346 // write to respective component tensor
347 tensorIndices[component].push_back(componentIndex);
348 tensorValues[component].push_back(elements[k].second);
349 }
350
351 // write data of each tensor
352 for (size_t i(0); i < componentTensors.size(); ++i) {
353 tensorIndices[i].reserve(tensorIndices[i].size()+1);
354 tensorValues[i].reserve(tensorIndices[i].size()+1);
355 get(i)->write(
356 tensorIndices[i].size(),
357 reinterpret_cast<int64_t *>(tensorIndices[i].data()),
358 tensorValues[i].data()

568 FockVectorNdCanonical(const unsigned int No, const unsigned int Nv) {
569 if (N > 6) {
570 throw new EXCEPTION("FockVectorNdCanonical implemented only up to 6");
571 }
572 const std::string pindices("abcdefg");
573 const std::string hindices("ijklomn");
574 for (unsigned int i(StartDimension); i <= N/2; i++) {
575
576 // vec<int> and not unsigned int, otherwise you'll be miserable for at
577 // least two days.
578 std::vector<int> syms(i, NS);
579 std::vector<int> dimso(i, No);
580 std::vector<int> dimsv(i, Nv);
581 std::vector<int> dims(dimsv);
582 dims.insert(dims.end(), dimso.begin(), dimso.end());
583
584 this->componentIndices.push_back(
585 pindices.substr(0, i) + hindices.substr(0, i)
586 );
587 this->componentTensors.push_back(
588 NEW(Tensor<F>, 2*i, dims.data(), syms.data(), *Sisi4s::world)
589 );
590 }
591 this->buildIndexTranslation();
592 }
