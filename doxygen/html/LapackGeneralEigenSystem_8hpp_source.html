215 double rightEigenError(const LapackMatrix<complex> &A) {
216 double error(0);
217 for (int i(0); i < A.getRows(); ++i) {
218 for (int k(0); k < A.getRows(); ++k) {
219 complex element(0);
220 for (int j(0); j < A.getRows(); ++j) {
221 element += A(i,j) * (*R)(j,k);
222 }
223 element -= lambdas[k] * (*R)(i,k);
224 error += std::real(element*std::conj(element));
225 }
226 }
227 return error;
228 }
229 double leftEigenError(const LapackMatrix<complex> &A) {
230 double error(0);
231 for (int j(0); j < A.getRows(); ++j) {
232 for (int k(0); k < A.getRows(); ++k) {
233 complex element(0);
234 for (int i(0); i < A.getRows(); ++i) {
235 element += std::conj((*L)(i,k)) * A(i,j);
236 }
237 element -= std::conj((*L)(j,k)) * lambdas[k];
238 error += std::real(element*std::conj(element));
239 }
240 }
241 return error;
242 }
243
244 double biorthogonalError() {
245 double error(0);
246 for (int i(0); i < R->getRows(); ++i) {
247 for (int j(0); j < R->getRows(); ++j) {
248 complex element(0);
249 for (int k(0); k < R->getRows(); ++k) {
250 element += std::conj((*R)(i,k)) * (*R)(j,k);
251 }
252 element -= i == j ? 1.0 : 0.0;
253 error += i == j ? 0.0 : std::real(element*std::conj(element));
254 }
255 }
256 return error;
