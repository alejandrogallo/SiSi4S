23 void refillBuffer() {
24 if (pos + REFILL_SIZE > buffer + BUFFER_SIZE && !eof) {
25 // move remaining text to beginning of buffer
26 std::memmove(buffer, pos, end-pos);
27 // set the buffer end to the end of the remaining text
28 end = buffer + (end - pos);
29 // the next character to read is now at the beginning
30 pos = buffer;
31 // fill the rest of the buffer from the file
32 int64_t size(buffer+BUFFER_SIZE-end);
33 int64_t count(stream->read(end, size).gcount());
34#ifdef DEBUG
35 LOG(2, "Scanner") << count << " bytes fetched." << std::endl;
36#endif
37 // account the read characters to the buffer
38 end += count;
39 // if not all requested characters could be read the file is done
40 if (count < size) {
41 eof = true;
42 *end = 0;
43 }
44 }
45 }
